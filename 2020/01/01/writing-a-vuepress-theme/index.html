# 数据响应式原理

vue3.0实现数据响应式的原理是采用了ES6的Proxy对象实现的。 使用Proxy的原因：

  • defineProperty只能监听某个属性，不能对对象监听
  • 可以省去 for in 提升效率
  • 可以监听数组，不用再去单独对数组做特异性操作
<!DOCTYPE HTML>
<html>
<head>
	<meta charset="utf-8" />
</head>
<body>
<div id="app"></div>
<script src="vue.js"></script>
<script>
var vm = new vue();
setTimeout(function(){
	vm.$data.a = '哈哈'
}, 2000);
</script>
</body>
</html>

vue.js:

function vue(){
	this.$data = {
		a: 1
	};
	this.el = document.getElementById('app');
	this._html = '';
	this.observe(this.$data);
	this.render();
}

vue.prototype.observe = function(obj){
	var self = this;
	this.$data = new Proxy(this.$data, {
		get: function(target, key){
			//vue在此进行依赖收集
			return target[key];
		},
		set: function(target, key, newVal) {
			target[key] = newVal;
			self.render();
		}
	});
}

vue.prototype.render = function() {
	this._html = 'i am a div, my value is ' + this.$data.a;
	this.el.innerHTML = this._html;
}
