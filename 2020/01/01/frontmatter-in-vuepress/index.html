# Js中的this用法

Js中的this用法详解

# 默认绑定全局变量

当函数被单独定义和调用的时候，应用的规则就是绑定全局变量。

function fn(){
  console.log(this);  //Window {postMessage: ƒ, blur: ƒ, focus: ƒ, close: ƒ, frames: Window, …}
  console.log(this.a);//2
}
var a = 2;
fn();

# 隐式绑定

函数调用时拥有一个上下文对象，就好像这个函数是属于该对象的一样。

function fnn(){
  console.log(this); //{a: 5, fn: ƒ fnn2()}
  console.log(this.a); //5
}
var obj = {
  a: 5,
  fn: fnn
}
obj.fn();

最后一个调用该函数的对象是传到函数的上下文对象：

function fnn(){
  console.log(this); //{a: 42, fn: ƒ fnn2()}
  console.log(this.a); //42
}
var obj2 = {
  a: 42,
  fn: fnn
};
var obj1 = {
  a: 2,
  obj2: obj2
};
obj1.obj2.fn();

失去隐式绑定的情况：

function fnn2(){
  console.log(this); //Window {postMessage: ƒ, blur: ƒ, focus: ƒ, close: ƒ, frames: Window, …}
  console.log(this.a); //全局
}
var obj = {
  a: 42,
  fn: fnn2
};
var bar = obj.fn; // 函数引用传递
var a = "全局";
bar();

# 显示绑定

bind()、apply()和call()，接收的第一个参数即是上下文对象并将其 赋给this。

function fn(){
  console.log(this); //{a: 10}
  console.log(this.a); //10
}
var obj = {
  a: 10
}
fn.call(obj);

如果我们传递第一个值为简单值，那么后台会自动转换为对应的封装对象：

function fn(){
  console.log(this); //Number {5}
  console.log(this.a); //undefined
}
fn.call(5);

如果传递为null或者不填，那么结果就是在绑定默认全局变量：

function fn(){
  console.log(this); //Window {postMessage: ƒ, blur: ƒ, focus: ƒ, close: ƒ, frames: Window, …}
  console.log(this.a); //100
}
var a = 100;
fn.call(null);

# new新对象绑定

如果是一个构造函数，那么用new来调用，那么绑定的将是新创建的对象：

function dog(name){
  console.log(this); //dog {}
  this.name = name;
  console.log(this); //dog {name: "柯基"}
}
var kj = new dog("柯基");
console.log(kj.name); //柯基
# 例子
function foo() {
  console.log(this.a);
}
var obj2 = {
  a: 42,
  foo: foo
};
var obj1 = {
  a: 2,
  obj2: obj2
};
obj1.obj2.foo(); //42	
function foo(something) {
  this.a = something;
}
var obj1 = {
  foo: foo
};
var obj2 = {};
obj1.foo(2);
console.log(obj1.a);     //2
obj1.foo.call(obj2, 3);
console.log(obj2.a);     //3
var bar = new obj1.foo(4);
console.log(obj1.a);     //2
console.log(bar.a);      //4
