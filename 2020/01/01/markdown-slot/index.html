# 数据响应式原理

vue2.0实现数据响应式的原理就是利用了Object.defineProperty()这个方法重新定义了对象获取属性值(get)和设置属性值(set)的操作来实现的。

<div id="app">
	<h1>hello</h1>
	<div>
	    <div v-text="myText"></div>
	    <input type="text" v-model="myText"/>
	    <div v-text="myBox"></div>
	    <input type="text" v-model="myBox"/>
	</div>
</div>
<script>
//发布者
class Vue {
	constructor(options){
	    //this是Vue实例对象
	    //获取数据
	    this.$data = options.data;
	    //获取元素对象 
	    this.$el = document.querySelector(options.el);
	    //发布者里有个容器,用来保存订阅者信息
	    //格式:{myText: [订阅者1, 订阅者2, ...], myText: [订阅者3, 订阅者4, ...], ...}
	    this._directive = {};
		
	    this.Observer(this.$data);
	    this.Compile(this.$el);
	}
	//主要功能是劫持数据
	Observer(data){
	    //key是myText和myBox
	    for(let key in data){
		    this._directive[key] = [];
		    let value = data[key];
		    let watcher = this._directive[key];
    		//第一个参数是需要修改的对象
    		//第二个参数是需要修改的属性
    		//第三个参数是需要修改的方法
    		let self = this;
		    Object.defineProperty(this.$data, key, {
		        get: function(){
			        return value;
		        },
               //执行self.$data[attrValue] = node.value;会触发set方法
		        //node.value的值会给newVal
		        set: function(newVal){
		        	if(newVal !== value){
    			        //把新值赋值给老值
                        //因为update()方法会触发get方法,get方法返回的value值要更新成新值
    			        value = newVal;
        			    //element是每一个订阅者的实例对象
        			    watcher.forEach(element => {
        				    element.update();
        			    });
			        }
		        }
		    });
	    }
	}
	//主要功能是解析指令
	//解析指令的目的:依赖收集
	//依赖收集后要更新视图
	Compile(el){
	    //获取该对象下所有子对象,不是所有后代对象,只是子对象
	    //第一次进来结果是数组[h1, div],里面是DOM对象
	    let nodes = el.children;
	    for(let i = 0; i < nodes.length; i++){
    		let node = nodes[i];
    		//递归查找子元素
    		if(node.children.length){
    		    this.Compile(node);
    		}
    		if(node.hasAttribute("v-text")){
    		    //订阅
    		    //attrValue是"myText"或"myBox"
    		    let attrValue = node.getAttribute("v-text");
    		    //div.innerHTML
    		    this._directive[attrValue].push(new Watcher(node, this, attrValue, "innerHTML"));
    		}
    		if(node.hasAttribute("v-model")){
    		    //订阅
    		    //attrValue是"myText"或"myBox"
    		    let attrValue = node.getAttribute("v-model");
    		    //input.value
    		    this._directive[attrValue].push(new Watcher(node, this, attrValue, "value"));
    		    let self = this;
    		    //监听文本框事件变化
    		    //input事件是元素获取用户输入时触发的
    		    node.addEventListener("input", function(){
        			//node.addEventListener里面的this指向的是node对象
        			//所以如果需要vue对象就要在外面引用一下
        			//也可以使用箭头函数,箭头函数的this是在定义函数时绑定的
        			//更新数据模型
        			self.$data[attrValue] = node.value;
    		    });
    		    //也可以使用bind方法来指定this对象
    		    /*
    		    node.addEventListener("input", (function(){
    			return function(){
    			    this.$data[attrValue] = node.value;
    			}
    		    })().bind(this));
    		    */
    		}
	    }
	}	
}

//订阅者,主要功能是更新视图
class Watcher {
	//el是DOM对象,vm是vue实例对象
	//exp是v-model或v-text里的值,也就是"myText"或"myBox"
	//attr是DOM对象要修改的属性,如"innerHTML"、"value"
	constructor(el, vm, exp, attr){
	    //this是watcher实例对象
	    this.el = el;
	    this.vm = vm;
	    this.exp = exp;
	    this.attr = attr;
	    this.update();
	}
	//更新视图
	update(){
        //会触发this.vm.$data[this.exp]的get方法
	    this.el[this.attr] = this.vm.$data[this.exp];
	}
}

const vm = new Vue({
	el: "#app",
	data: {
	    myText: "whz",
	    myBox: "box"
	}
});
</script>
